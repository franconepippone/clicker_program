; Basic arithmetics
; N.B: SPACES are important! x=3 doesn't work

var y = 4

var out = x + y

var out = x - y

var out = x * y
printvar out

var out = x / y
printvar out


; Loops

var i = 0
label loopStart
	var i = i + 1
	printvar i
	wait .2
	jump loopStart 10
