;; EXAMPLE TO SHOW USAGE OF VARIABLES
;; example_programs/variables.txt

;; Basic arithmetics
; N.B: SPACES are important! x=3 doesn't work
var x = 3 
var y = 4

var out = x + y	 
printvar out

var out = x - y
printvar out

var out = x * y
printvar out

var out = x / y
printvar out

wait 1
pause	; give time to observe results

; Loops

var i = 0
label loopStart
	var i = i + 1
	printvar i
	wait .2
	jump loopStart 10


;; Combining variables and move commands

; Moves the mouse in a parabula shape
var mouse_x = $MOUSE_X ; initializing variables
var mouse_y = $MOUSE_Y
var y_vel = -30

label drawStart
	var mouse_x = mouse_x + 10
	var mouse_y = mouse_y + y_vel
	var y_vel = y_vel + 2		; Decreases velocity, acts like gravity
	
	move mouse_x mouse_y
	jump drawStart 40
