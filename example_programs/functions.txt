;; EXAMPLE TO SHOW THE WORKINGS OF CALL / RETURN
;; example_programs/functions.txt

jump START_SCRIPT ; Jump to the main portion of the script

;; This function moves the mouse in a square
label function_1
	moverel 50 0 1
	moverel 0 50 1
	moverel -50 0 1
	moverel 0 -50 1
	; return here is mandatory if we plan on doing "call function_1"
	return	; Here the program jumps to line 24, right after "call ..."



label START_SCRIPT ; <-- script starts here
	print Starting program!
	centermouse	
	call function_1	; Program jumps to function_1 label
	print Program ended!
